import java.util.Scanner; public class Main { public static void main(String[] args) { Scanner input = new Scanner(System.in); System.out.println("=================================="); System.out.println("3 SAYILI HARMONİK HESAP MAKİNASI"); System.out.println("=================================="); System.out.print("İlk Harmonik Sayıyı Giriniz : "); int number = input.nextInt(); System.out.print("İkinci Harmonik Sayıyı Giriniz : "); int number1 = input.nextInt(); System.out.print("Üçüncü Harmonik Sayıyı Giriniz : "); int number2 = input.nextInt(); int[] harmonic = {number,number1,number2}; double result=0.0; double avarage; for (int i=0; i< harmonic.length; i++){ result += 1.0/harmonic[i]; } avarage = harmonic.length/result; System.out.println("=================================="); System.out.println("Harmonik Ortalamanız : "+avarage); System.out.println("=================================="); } }
